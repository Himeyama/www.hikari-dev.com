C 言語で 2 次元配列を扱う

2022-01-15

行数・列数、数値の入るメモリを構造体でまとめると扱いやすい。

#include<stdio.h>#include<stdlib.h>typedefstruct{float*data;intcol_size;introw_size;}Mat;voidMatInit(Mat*mat,introw_size,intcol_size){mat->row_size=row_size;mat->col_size=col_size;mat->data=(float*)calloc(row_size*col_size,sizeof(float));}float*MatAt(Mat*mat,inti,intj){returnmat->data+i*mat->col_size+j;}voidMatFree(Mat*mat){free(mat->data);}intmain(void){Matmat;MatInit(&mat,30,5);// 30 行 5 列 の行列を初期化*MatAt(&mat,0,0)=50;// 行 0 列 0 に 50 を代入printf("%f\n",*MatAt(&mat,0,0));// 行 0, 列 0 の数値を表示MatFree(&mat);return0;}
